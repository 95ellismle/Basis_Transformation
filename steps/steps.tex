\begin{dmath*}
\sum\limits_{k,z,n,m}  U_{l,k} U_{m,k}U_{m,z}U_{n,z} u_{n}+
\sum\limits_{k,z,n,m}   u_{k} +

\end{dmath*}

\begin{dmath*}

\sum\limits_{k,z,n,m}   u_{k} +
\sum\limits_{k}   u_{k} U_{p,k}U_{s,k}
\end{dmath*}
